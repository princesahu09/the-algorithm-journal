class Solution {
public:
    int sumOfUnique(vector<int>& nums) {
        unordered_map<int,int>ans;
        for(int i=0;i<nums.size();i++){
            ans[nums[i]]++;
        }
        int sum=0;
        for(auto & x:ans){
            if(x.second==1){
                sum+=x.first;
            }
        }
        return sum;
    }
};
