class Solution {
public:
    bool isSameAfterReversals(int x) {
        int original = x;

        int ans = 0;
        while (x != 0) {
            int rem = x % 10;
            ans = ans * 10 + rem;
            x = x / 10;
        }

        int ans2 = 0;
        while (ans != 0) {
            int rem2 = ans % 10;
            ans2 = ans2 * 10 + rem2;
            ans = ans / 10;
        }
        if (ans2 == original)
            return true;

        return false;
    }
};
