class Solution {
public:
    vector<bool> prefixesDivBy5(vector<int>& nums) {

        vector<bool> ans;
        int value = 0;

        for(int i = 0; i < nums.size(); i++) {
            value = (value * 2 + nums[i]) % 5;
            if(value == 0){
                ans.push_back(true);
            }
            else
            {
                ans.push_back(false);
            }
        }
        return ans;
    }
};
