class Solution {
public:
    int getLucky(string s, int k) {
        string ans = "";
        for(int i=0;i<s.size();i++){
            int val = (s[i]-'a')+1;
            ans+=to_string(val);
        }
        int sum = 0;
        while(k--){
            sum = 0;
        for(int i=0;i<ans.size();i++){
            sum += ans[i]-'0';
          }
          ans = to_string(sum);
        }
        return sum;
    }
};
