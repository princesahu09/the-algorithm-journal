class Solution {
public:
    char kthCharacter(int k) {
        string s = "a";
        if (k == 1){
       return 'a';
        } 

        while (s.size() < k) {
            string x = "";
            for (int i = 0; i < s.size(); i++) {

                x += char(s[i] + 1);   
            }
            s += x;   
        }

        return s[k-1];  
    }
};
