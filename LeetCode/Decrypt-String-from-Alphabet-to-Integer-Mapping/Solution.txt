func freqAlphabets(s string) string {
    ans := []rune{}
	n := len(s)

	for i := 0; i < n; {
		if i+2 < n && s[i+2] == '#' {
			// take two digits before '#'
			num := (int(s[i]-'0')*10 + int(s[i+1]-'0'))
			ans = append(ans, rune('a'+num-1))
			i += 3 // skip processed part "xx#"
		} else {
			// single digit case
			num := int(s[i] - '0')
			ans = append(ans, rune('a'+num-1))
			i++
		}
	}

	return string(ans)
    
}
