class Solution {
public:
    long long sumAndMultiply(int n) {
        string s=to_string(n);
        long long  sum=0;
        string t="";
        for(int i=0;i<s.size();i++){
            if(s[i]!='0'){
            sum+=s[i]-'0';
            t+=s[i];
        }
        }
        if(t==""){
            return 0;
        }
        long long p=stoll(t);
        return sum*p;
        
        
    }
};
