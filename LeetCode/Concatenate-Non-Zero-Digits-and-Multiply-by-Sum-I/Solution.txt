class Solution {
public:
    long long sumAndMultiply(int n) {


        long long sum=0;
        long long number=0;
        while(n>0)
            {
                int r=n%10;
                if(r!=0)
                {
                    number=number*10+r;
                }
                sum+=r;
                n/=10;
            }
        string s=to_string(number);
        reverse(s.begin(),s.end());
        number=stoll(s);
        return sum*number;
        
    }
};
