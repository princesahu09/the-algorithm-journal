1class Solution {
2public:
3    int largestPerimeter(vector<int>& nums) {
4        int n = nums.size();
5        sort(nums.begin(),nums.end());
6        for(int i=n-1;i>=2;i--){
7            int a = nums[i-2];

12        }
13        return 0;
14    }
