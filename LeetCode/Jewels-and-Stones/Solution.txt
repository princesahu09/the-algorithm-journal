class Solution {
public:
    int numJewelsInStones(string jewels, string stones) {
        unordered_map<int,int>mp;
        for(int i=0;i<stones.size();i++){
            mp[stones[i]]++;
        }
        int count = 0;
        for(char c : jewels){
            if(mp.find(c)!=mp.end()){
                count+=mp[c];
            }
        }
        return count;
    }
};
