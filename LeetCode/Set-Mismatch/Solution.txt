class Solution {
public:
    vector<int> findErrorNums(vector<int>& nums) {
        int n = nums.size();
        unordered_map<int,int> ans;

        for(int i=0;i<nums.size();i++){
            ans[nums[i]]++;
        }

       
        int x = -1 ;
        int y=0;

        for(int i = 1; i <= n; i++) {
            if (ans[i] == 2)
                x = i;
            else if (ans[i] == 0)
                y = i;
        }

        return {x, y};
        
    }
};
