class Solution {
public:
    bool uniqueOccurrences(vector<int>& arr) {
        unordered_map<int,int>mp;
        for(int i=0;i<arr.size();i++){
            mp[arr[i]]++;
        }

       unordered_set<int>freq;
       for(auto& it : mp){
          freq.insert(it.second);
       }
       if(freq.size() == mp.size()) return true;

       return false;
    }
};
