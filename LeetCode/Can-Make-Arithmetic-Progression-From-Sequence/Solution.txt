class Solution {
public:
    bool canMakeArithmeticProgression(vector<int>& arr) {
        sort(arr.begin(),arr.end());
        vector<int>ans;
        for(int i=1;i<arr.size();i++){
            int a = abs(arr[i]-arr[i-1]);
            ans.push_back(a);
        }
       for(int i=1;i<ans.size();i++){
          if(ans[i]!=ans[i-1])
          return 0;
       }
       return 1;
    }
};
