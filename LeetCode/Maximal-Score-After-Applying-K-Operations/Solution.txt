class Solution {
public:
    long long maxKelements(vector<int>& nums, int k) {
        priority_queue<int>p;
        for(int i=0;i<nums.size();i++){
            p.push(nums[i]);
        }
        long long sum = 0;

        while(k){
            int maxi = p.top();
            p.pop();

            sum += maxi;
            
            int num = (maxi+2)/3;
            p.push(num);

            k--;
        }
        return sum;
    }
};
