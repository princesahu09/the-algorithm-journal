class Solution {
public:
    vector<int> findDisappearedNumbers(vector<int>& nums) {
        unordered_set<int>ans(nums.begin(),nums.end());

        vector<int>res;
        int n=nums.size();

        for(int i=1;i<=n;i++){
            if(ans.count(i)){
                continue;
            }else{
                res.push_back(i);
            }
        }
        return res;
    }
};
